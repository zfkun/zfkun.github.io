指令

# 下载视频和音频
youtube-dl -f bestvideo+bestaudio "https://www.youtube.com/watch?v=MLJBNovl1RQ"

# 下载自动生成字幕 (vtt)
youtube-dl --write-auto-sub "https://www.youtube.com/watch?v=MLJBNovl1RQ"

# 下载自带字幕 (vtt)
youtube-dl --write-sub "https://www.youtube.com/watch?v=MLJBNovl1RQ"

# 查看字幕列表
youtube-dl --list-subs "https://www.youtube.com/watch?v=MLJBNovl1RQ"
